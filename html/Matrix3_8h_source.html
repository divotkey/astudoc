346  Matrix3<T>& Invert() {
347  T tmp[9];
348 
349  tmp[0] = m[4] * m[8] - m[5] * m[7];
350  tmp[1] = m[7] * m[2] - m[8] * m[1];
351  tmp[2] = m[1] * m[5] - m[2] * m[4];
352  tmp[3] = m[5] * m[6] - m[3] * m[8];
353  tmp[4] = m[0] * m[8] - m[2] * m[6];
354  tmp[5] = m[2] * m[3] - m[0] * m[5];
355  tmp[6] = m[3] * m[7] - m[4] * m[6];
356  tmp[7] = m[6] * m[1] - m[7] * m[0];
357  tmp[8] = m[0] * m[4] - m[1] * m[3];
358 
359  T determinant = m[0] * tmp[0] + m[1] * tmp[3] + m[2] * tmp[6];
360  if (abs(determinant) < std::numeric_limits<T>::min()) {
361  SetToIdentity();
362  }
363 
364  T invDeterminant = 1 / determinant;
365  m[0] = invDeterminant * tmp[0];
366  m[1] = invDeterminant * tmp[1];
367  m[2] = invDeterminant * tmp[2];
368  m[3] = invDeterminant * tmp[3];
369  m[4] = invDeterminant * tmp[4];
370  m[5] = invDeterminant * tmp[5];
371  m[6] = invDeterminant * tmp[6];
372  m[7] = invDeterminant * tmp[7];
373  m[8] = invDeterminant * tmp[8];
374 
375  return *this;
376  }
