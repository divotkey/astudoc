82  ColorHsv & Set(const Color4d & c) {
83  double min = std::min({c.r, c.g, c.b});
84  double max = std::max({c.r, c.g, c.b});
85  double delta = max - min;
86 
87  this->v = max;
88  if (max != 0) {
89  this->s = delta / max;
90  }
91  else {
92  // r = g = b = 0
93  this->s = 0;
94  this->h = -1;
95  return *this;
96  }
97 
98  if (max == min) {
99  // Everything is gray.
100  this->h = 0;
101  this->s = 0;
102  return *this;
103  }
104 
105  if (c.r == max) {
106  // Between Yellow and Magenta.
107  this->h = (c.g - c.b ) / delta;
108  }
109  else if(c.g == max ) {
110  // Between Cyan and Yellow.
111  this->h = 2 + (c.b - c.r) / delta;
112  }
113  else {
114  // Between Magenta and Cyan
115  this->h = 4.0 + (c.r - c.g) / delta;
116  }
117 
118  // Make Degrees.
119  this->h *= 60;
120 
121  if( this->h < 0 ) {
122  this->h += 360;
123  }
124 
125  return *this;
126  }

133  Color4d ToRgb() const {
134  int i;
135  if (s == 0) {
136  // Achromatic (Gray).
137  return Color4d(v, v, v);
138  }
139 
140  // Sector 0 to 5.
141  double h = this->h / 60.0;
142 
143  i = static_cast<int>(floor(h));
144 
145  // Factorial part of h.
146  double f = h - i;
147 
148  double p = v * (1.0 - s );
149  double q = v * (1.0 - s * f);
150  double t = v * ( 1 - s * ( 1 - f ) );
151 
152  switch( i ) {
153  case 0:
154  // *r = v; *g = t; *b = p;
155  return Color4d(v, t, p);
156 
157  case 1:
158  //*r = q; *g = v; *b = p;
159  return Color4d(q, v, p);
160 
161  case 2:
162  // *r = p; *g = v; *b = t;
163  return Color4d(p, v, t);
164 
165  case 3:
166  // *r = p; *g = q; *b = v;
167  return Color4d(p, q, v);
168 
169  case 4:
170  // *r = t; *g = p; *b = v;
171  return Color4d(t, p, v);
172 
173  case 5:
174  // *r = v; *g = p; *b = q;
175  return Color4d(v, p, q);
176 
177  default:
178  // Should never happen.
179  throw std::logic_error("Conversion of HSV color to RGB failed");
180  }
