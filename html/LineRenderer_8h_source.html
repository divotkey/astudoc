192  virtual void DrawCircle(T cx, T cy, T r, unsigned int segments = 24) {
193  T da = static_cast<T>(MathUtils::PI2d) / segments;
194 
195  Vector2<T> p0(0, r);
196  for (unsigned int i = 0; i < segments; ++i) {
197  Vector2<T> p1(0, r);
198  p1.Rotate(i * da);
199 
200  DrawLine(p0.x + cx, p0.y + cy, p1.x + cx, p1.y + cy);
201  p0 = p1;
202  }
203  DrawLine(p0.x + cx, p0.y + cy, cx, r + cy);
204  }
