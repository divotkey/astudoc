136  virtual T Update(T dt) override {
137  // Calc difference (error) between current value and target value.
138  const T e = Interpolator1<T>::targetValue - Interpolator1<T>::currentValue;
139 
140  // Determine velocity of value change.
141  float v = (e == 0.0) ? 0.0f : std::copysignf(speed, e);
142 
143  // Update and current value.
144  Interpolator1<T>::currentValue += v * dt;
145 
146  // Test is error has changed sign. If that happens we did overshoot
147  // the target value.
148  if ((Interpolator1<T>::targetValue - Interpolator1<T>::currentValue) * e < 0) {
149  Interpolator1<T>::currentValue = Interpolator1<T>::targetValue;
150  }
151

153  }
